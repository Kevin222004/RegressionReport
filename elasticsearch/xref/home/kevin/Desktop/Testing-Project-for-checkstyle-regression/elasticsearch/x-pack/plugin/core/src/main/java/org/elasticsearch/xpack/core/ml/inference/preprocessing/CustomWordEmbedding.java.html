36  /**
37   * This is a pre-processor that embeds text into a numerical vector.
38   *
39   * It calculates a set of features based on script type, ngram hashes, and most common script values.
40   *
41   * The features are then concatenated with specific quantization scales and weights into a vector of length 80.
42   *
43   * This is a fork and a port of: https://github.com/google/cld3/blob/06f695f1c8ee530104416aab5dcf2d6a1414a56a/src/embedding_network.cc
44   */
45  public class CustomWordEmbedding implements LenientlyParsedPreProcessor, StrictlyParsedPreProcessor {

73      private static final ConstructingObjectParser<CustomWordEmbedding, PreProcessorParseContext> STRICT_PARSER = createParser(false);

76      @SuppressWarnings("unchecked")
77      private static ConstructingObjectParser<CustomWordEmbedding, PreProcessorParseContext> createParser(boolean lenient) {
78          ConstructingObjectParser<CustomWordEmbedding, PreProcessorParseContext> parser = new ConstructingObjectParser<>(
79              NAME.getPreferredName(),
80              lenient,
81              (a, c) -> new CustomWordEmbedding((short[][]) a[0], (byte[][]) a[1], (String) a[2], (String) a[3])
82          );
83  
84          parser.declareField(ConstructingObjectParser.constructorArg(), (p, c) -> {
85              List<List<Short>> listOfListOfShorts = MlParserUtils.parseArrayOfArrays(
86                  EMBEDDING_QUANT_SCALES.getPreferredName(),
87                  XContentParser::shortValue,
88                  p
89              );
90              short[][] primitiveShorts = new short[listOfListOfShorts.size()][];
91              int i = 0;
92              for (List<Short> shorts : listOfListOfShorts) {
93                  short[] innerShorts = new short[shorts.size()];
94                  for (int j = 0; j < shorts.size(); j++) {
95                      innerShorts[j] = shorts.get(j);
96                  }
97                  primitiveShorts[i++] = innerShorts;
98              }
99              return primitiveShorts;
100         }, EMBEDDING_QUANT_SCALES, ObjectParser.ValueType.VALUE_ARRAY);
101         parser.declareField(ConstructingObjectParser.constructorArg(), (p, c) -> {
102             List<byte[]> values = new ArrayList<>();
103             while (p.nextToken() != XContentParser.Token.END_ARRAY) {
104                 values.add(p.binaryValue());
105             }
106             byte[][] primitiveBytes = new byte[values.size()][];
107             int i = 0;
108             for (byte[] bytes : values) {
109                 primitiveBytes[i++] = bytes;
110             }
111             return primitiveBytes;
112         }, EMBEDDING_WEIGHTS, ObjectParser.ValueType.VALUE_ARRAY);
113         parser.declareString(ConstructingObjectParser.constructorArg(), FIELD);
114         parser.declareString(ConstructingObjectParser.constructorArg(), DEST_FIELD);
115         return parser;
116     }

126     private static final int CONCAT_LAYER_SIZE = 80;
127     private static final int[] EMBEDDING_DIMENSIONS = new int[] { 16, 16, 8, 8, 16, 16 };
128 
129     // Order matters
130     private static final List<FeatureExtractor> FEATURE_EXTRACTORS = Arrays.asList(
131         new NGramFeatureExtractor(2, 1000),
132         new NGramFeatureExtractor(4, 5000),
133         new RelevantScriptFeatureExtractor(),
134         new ScriptFeatureExtractor(),
135         new NGramFeatureExtractor(3, 5000),
136         new NGramFeatureExtractor(1, 100)
137     );

140     private final byte[][] embeddingsWeights;
141     private final String fieldName;

144     public CustomWordEmbedding(StreamInput in) throws IOException {
145         this.fieldName = in.readString();
146         this.destField = in.readString();
147         this.embeddingsWeights = in.readArray(StreamInput::readByteArray, (length) -> new byte[length][]);
148         this.embeddingsQuantScales = in.readArray((input -> {
149             int length = input.readVInt();
150             short[] shorts = new short[length];
151             for (int i = 0; i < length; i++) {
152                 shorts[i] = in.readShort();
153             }
154             return shorts;
155         }), (length) -> new short[length][]);
156     }

158     public CustomWordEmbedding(short[][] embeddingsQuantScales, byte[][] embeddingsWeights, String fieldName, String destField) {
159         this.embeddingsQuantScales = embeddingsQuantScales;
160         this.embeddingsWeights = embeddingsWeights;
161         this.fieldName = fieldName;
162         this.destField = destField;
163     }

165     /**
166      * Derived from: https://github.com/google/cld3/blob/06f695f1c8ee530104416aab5dcf2d6a1414a56a/src/embedding_network.cc#L74
167      */
168     private double[] concatEmbeddings(List<FeatureValue[]> featureVectors) {
169         double[] concat = new double[CONCAT_LAYER_SIZE];
170 
171         int offset = 0;
172         // "esIndex" stands for "embedding space index".
173         for (int esIndex = 0; esIndex < featureVectors.size(); ++esIndex) {
174             byte[] embeddingWeight = this.embeddingsWeights[esIndex];
175             short[] quants = this.embeddingsQuantScales[esIndex];
176             int embeddingDim = EMBEDDING_DIMENSIONS[esIndex];
177 
178             FeatureValue[] featureVector = featureVectors.get(esIndex);
179             assert (offset + embeddingDim <= concat.length);
180             for (FeatureValue featureValue : featureVector) {
181                 // Multiplier for each embedding weight.
182                 int row = featureValue.getRow();
183                 double multiplier = featureValue.getWeight() * shortToDouble(quants[row]);
184 
185                 // Iterate across columns for this row
186                 for (int i = 0; i < embeddingDim; ++i) {
187                     double value = (getRowMajorData(embeddingWeight, embeddingDim, row, i)) * multiplier;
188                     int concatIndex = offset + i;
189                     concat[concatIndex] += value;
190                 }
191             }
192             offset += embeddingDim;
193         }
194 
195         return concat;
196     }

198     /**
199      * Internal logic to CLD3. The quantiles are stored as "compact float16" values.
200      * These values are stored as `unsigned int16` in C++, best represented by a `short` Java
201      * To "expand" these to a regular float, the mantissa bytes are filled with 0, a la a bit shift of 16
202      * The memory is then statically cast to a float with the shift.
203      *
204      * The following satisfies this in java
205      * @param s Represents a "compact float16", i.e. unsigned int16
206      * @return Float representation with the short shifted over to zero out the underlying mantissa bytes
207      */
208     private static double shortToDouble(short s) {
209         return Float.intBitsToFloat(s << 16);
210     }
211 
212     private static int getRowMajorData(byte[] data, int colDim, int row, int col) {
213         return data[row * colDim + col];
214     }

226     @Override
227     public void process(Map<String, Object> fields) {
228         Object field = fields.get(fieldName);
229         if ((field instanceof String) == false) {
230             return;
231         }
232         String text = (String) field;
233         text = FeatureUtils.cleanAndLowerText(text);
234         text = FeatureUtils.truncateToNumValidBytes(text, MAX_STRING_SIZE_IN_BYTES);
235         final String finalText = text;
236         if (finalText.isEmpty() || finalText.isBlank()) {
237             fields.put(destField, List.of());
238             return;
239         }
240         List<StringLengthAndEmbedding> embeddings = new ArrayList<>();
241         int[] codePoints = finalText.codePoints().toArray();
242         for (int i = 0; i < codePoints.length - 1;) {
243             while (i < codePoints.length - 1 && Character.isLetter(codePoints[i]) == false) {
244                 i++;
245             }
246             if (i >= codePoints.length) {
247                 break;
248             }
249             Character.UnicodeScript currentCode = Character.UnicodeScript.of(codePoints[i]);
250             int j = i + 1;
251             for (; j < codePoints.length; j++) {
252                 while (j < codePoints.length && Character.isLetter(codePoints[j]) == false) {
253                     j++;
254                 }
255                 if (j >= codePoints.length) {
256                     break;
257                 }
258                 Character.UnicodeScript j1 = Character.UnicodeScript.of(codePoints[j]);
259                 if (j1 != currentCode && j1 != Character.UnicodeScript.INHERITED) {
260                     if (j < codePoints.length - 1) {
261                         Character.UnicodeScript j2 = Character.UnicodeScript.of(codePoints[j + 1]);
262                         if (j2 != Character.UnicodeScript.COMMON && j2 != currentCode) {
263                             break;
264                         }
265                     }
266                 }
267             }
268             // Knowing the start and the end of the section is important for feature building, so make sure its wrapped in spaces
269             String str = new String(codePoints, i, j - i);
270             StringBuilder builder = new StringBuilder();
271             if (str.startsWith(" ") == false) {
272                 builder.append(" ");
273             }
274             builder.append(str);
275             if (str.endsWith(" ") == false) {
276                 builder.append(" ");
277             }
278             embeddings.add(
279                 new StringLengthAndEmbedding(
280                     // Don't count white spaces as bytes for the prediction
281                     // We ues utf-8 length here as
282                     // * The original C++ implementation does this when measuring string length
283                     // * Languages with complex characters (like zh) convey more information per a single utf-16 character and
284                     // using utf-8 length captures that.
285                     str.trim().getBytes(StandardCharsets.UTF_8).length,
286                     concatEmbeddings(
287                         FEATURE_EXTRACTORS.stream()
288                             .map((featureExtractor) -> featureExtractor.extractFeatures(builder.toString()))
289                             .collect(Collectors.toList())
290                     )
291                 )
292             );
293             i = j;
294         }
295         fields.put(destField, embeddings);
296     }

313     @Override
314     public long ramBytesUsed() {
315         long size = SHALLOW_SIZE;
316         for (byte[] bytes : embeddingsWeights) {
317             size += RamUsageEstimator.sizeOf(bytes);
318         }
319         for (short[] shorts : embeddingsQuantScales) {
320             size += RamUsageEstimator.sizeOf(shorts);
321         }
322         return size;
323     }

330     @Override
331     public void writeTo(StreamOutput out) throws IOException {
332         out.writeString(fieldName);
333         out.writeString(destField);
334         out.writeArray(StreamOutput::writeByteArray, embeddingsWeights);
335         out.writeArray((output, value) -> {
336             output.writeVInt(value.length);
337             for (short s : value) {
338                 output.writeShort(s);
339             }
340         }, embeddingsQuantScales);
341     }

359     @Override
360     public boolean equals(Object o) {
361         if (this == o) return true;
362         if (o == null || getClass() != o.getClass()) return false;
363         CustomWordEmbedding that = (CustomWordEmbedding) o;
364         return Objects.equals(fieldName, that.fieldName)
365             && Objects.equals(destField, that.destField)
366             && Arrays.deepEquals(embeddingsWeights, that.embeddingsWeights)
367             && Arrays.deepEquals(embeddingsQuantScales, that.embeddingsQuantScales);
368     }
