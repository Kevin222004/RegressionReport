8   package org.elasticsearch.xpack.security.authc.jwt;
9   
10  import com.nimbusds.jose.JOSEException;
11  import com.nimbusds.jose.jwk.JWK;
12  import com.nimbusds.jose.jwk.JWKSet;
13  
14  import org.apache.logging.log4j.LogManager;
15  import org.apache.logging.log4j.Logger;
16  import org.elasticsearch.xpack.core.security.authc.jwt.JwtRealmSettings;
17  import org.elasticsearch.xpack.core.security.user.User;
18  
19  import java.io.Closeable;
20  import java.io.IOException;
21  import java.nio.charset.StandardCharsets;
22  import java.util.Collections;
23  import java.util.List;
24  import java.util.Map;
25  
26  import static org.elasticsearch.test.ESTestCase.randomBoolean;
27  
28  /**
29   * Test class with settings for a JWT issuer to sign JWTs for users.
30   * Based on these settings, a test JWT realm can be created.
31   */
32  public class JwtIssuer implements Closeable {
33      private static final Logger LOGGER = LogManager.getLogger(JwtIssuer.class);
34  
35      record AlgJwkPair(String alg, JWK jwk) {}
36  
37      // input parameters
38      final String issuerClaimValue; // claim name is hard-coded to `iss` for OIDC ID Token compatibility
39      final List<String> audiencesClaimValue; // claim name is hard-coded to `aud` for OIDC ID Token compatibility
40      final String principalClaimName; // claim name is configurable, EX: Users (sub, oid, email, dn, uid), Clients (azp, appid, client_id)
41      final Map<String, User> principals; // principals with roles, for sending encoded JWTs into JWT realms for authc/authz verification
42      final JwtIssuerHttpsServer httpsServer;
43  
44      List<String> algorithmsAll;
45  
46      // Computed values
47      List<AlgJwkPair> algAndJwksPkc;
48      List<AlgJwkPair> algAndJwksHmac;
49      AlgJwkPair algAndJwkHmacOidc;
50      List<AlgJwkPair> algAndJwksAll;
51      String encodedJwkSetPkcPublicPrivate;
52      String encodedJwkSetPkcPublic;
53      String encodedJwkSetHmac;
54      String encodedKeyHmacOidc;
55  
56      JwtIssuer(
57          final String issuerClaimValue,
58          final List<String> audiencesClaimValue,
59          final String principalClaimName,
60          final Map<String, User> principals,
61          final boolean createHttpsServer
62      ) throws Exception {
63          this.issuerClaimValue = issuerClaimValue;
64          this.audiencesClaimValue = audiencesClaimValue;
65          this.principalClaimName = principalClaimName;
66          this.principals = principals;
67          this.httpsServer = createHttpsServer ? new JwtIssuerHttpsServer(null) : null;
68      }
69  
70      // The flag areHmacJwksOidcSafe indicates if all provided HMAC JWKs are UTF8, for HMAC OIDC JWK encoding compatibility.
71      void setJwks(final List<AlgJwkPair> algAndJwks, final boolean areHmacJwksOidcSafe) throws JOSEException {
72          this.algorithmsAll = algAndJwks.stream().map(e -> e.alg).toList();
73          LOGGER.info("Setting JWKs: algorithms=[{}], areHmacJwksOidcSafe=[{}]", String.join(",", this.algorithmsAll), areHmacJwksOidcSafe);
74          this.algAndJwksAll = algAndJwks;
75          this.algAndJwksPkc = this.algAndJwksAll.stream()
76              .filter(e -> JwtRealmSettings.SUPPORTED_SIGNATURE_ALGORITHMS_PKC.contains(e.alg))
77              .toList();
78          this.algAndJwksHmac = this.algAndJwksAll.stream()
79              .filter(e -> JwtRealmSettings.SUPPORTED_SIGNATURE_ALGORITHMS_HMAC.contains(e.alg))
80              .toList();
81          if ((this.algAndJwksHmac.size() == 1) && (areHmacJwksOidcSafe) && (randomBoolean())) {
82              this.algAndJwkHmacOidc = this.algAndJwksHmac.get(0);
83              this.algAndJwksHmac = Collections.emptyList();
84          } else {
85              this.algAndJwkHmacOidc = null;
86          }
87  
88          // Encode PKC JWKSet (key material bytes are wrapped in Base64URL, and then wraps in JSON)
89          final JWKSet jwkSetPkc = new JWKSet(this.algAndJwksPkc.stream().map(p -> p.jwk).toList());
90          this.encodedJwkSetPkcPublicPrivate = JwtUtil.serializeJwkSet(jwkSetPkc, false);
91          this.encodedJwkSetPkcPublic = JwtUtil.serializeJwkSet(jwkSetPkc, true);
92  
93          // Encode HMAC JWKSet (key material bytes are wrapped in Base64URL, and then wraps in JSON)
94          final JWKSet jwkSetHmac = new JWKSet(this.algAndJwksHmac.stream().map(p -> p.jwk).toList());
95          this.encodedJwkSetHmac = JwtUtil.serializeJwkSet(jwkSetHmac, false);
96  
97          // Encode HMAC OIDC JWK (key material bytes are decoded from UTF8 to UNICODE String)
98          this.encodedKeyHmacOidc = (algAndJwkHmacOidc == null) ? null : JwtUtil.serializeJwkHmacOidc(this.algAndJwkHmacOidc.jwk);
99  
100         if (this.httpsServer != null) {
101             this.httpsServer.updateJwkSetPkcContents(this.encodedJwkSetPkcPublic.getBytes(StandardCharsets.UTF_8));
102         }
103     }
104 
105     @Override
106     public void close() {
107         if (this.httpsServer != null) {
108             try {
109                 this.httpsServer.close();
110             } catch (IOException e) {
111                 LOGGER.warn("Exception closing HTTPS server for issuer [" + issuerClaimValue + "]", e);
112             }
113         }
114     }
115 }
