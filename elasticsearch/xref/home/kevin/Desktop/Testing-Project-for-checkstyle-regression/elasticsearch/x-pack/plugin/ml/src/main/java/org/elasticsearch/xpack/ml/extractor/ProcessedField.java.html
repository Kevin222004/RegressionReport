7   package org.elasticsearch.xpack.ml.extractor;
8   
9   import org.elasticsearch.common.util.Maps;
10  import org.elasticsearch.search.SearchHit;
11  import org.elasticsearch.xpack.core.ml.inference.preprocessing.PreProcessor;
12  
13  import java.util.Collections;
14  import java.util.List;
15  import java.util.Map;
16  import java.util.Objects;
17  import java.util.Set;
18  import java.util.function.Function;
19  
20  import static org.elasticsearch.xpack.ml.dataframe.extractor.DataFrameDataExtractor.isValidValue;
21  
22  public class ProcessedField {
23      private final PreProcessor preProcessor;
24  
25      public ProcessedField(PreProcessor processor) {
26          this.preProcessor = Objects.requireNonNull(processor);
27      }
28  
29      public List<String> getInputFieldNames() {
30          return preProcessor.inputFields();
31      }
32  
33      public List<String> getOutputFieldNames() {
34          return preProcessor.outputFields();
35      }
36  
37      public Set<String> getOutputFieldType(String outputField) {
38          return Collections.singleton(preProcessor.getOutputFieldType(outputField));
39      }
40  
41      public Object[] value(SearchHit hit, Function<String, ExtractedField> fieldExtractor) {
42          List<String> inputFields = getInputFieldNames();
43          Map<String, Object> inputs = Maps.newMapWithExpectedSize(inputFields.size());
44          for (String field : inputFields) {
45              ExtractedField extractedField = fieldExtractor.apply(field);
46              if (extractedField == null) {
47                  return new Object[0];
48              }
49              Object[] values = extractedField.value(hit);
50              if (values == null || values.length == 0) {
51                  continue;
52              }
53              final Object value = values[0];
54              if (values.length == 1 && (isValidValue(value))) {
55                  inputs.put(field, value);
56              }
57          }
58          preProcessor.process(inputs);
59          return preProcessor.outputFields().stream().map(inputs::get).toArray();
60      }
61  
62      public String getProcessorName() {
63          return preProcessor.getName();
64      }
65  
66  }
