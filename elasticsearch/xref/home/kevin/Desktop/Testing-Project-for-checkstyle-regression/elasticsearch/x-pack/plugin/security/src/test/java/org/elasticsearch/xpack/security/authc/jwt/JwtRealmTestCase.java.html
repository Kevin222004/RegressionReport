128     protected List<JwtIssuerAndRealm> generateJwtIssuerRealmPairs(
129         final JwtRealmsServiceSettingsBuilder jwtRealmsServiceSettingsBuilder,
130         final MinMax realmsRange,
131         final MinMax authzRange,
132         final MinMax algsRange,
133         final MinMax audiencesRange,
134         final MinMax usersRange,
135         final MinMax rolesRange,
136         final MinMax jwtCacheSizeRange,
137         final boolean createHttpsServer
138     ) throws Exception {
139         assertThat(realmsRange.min(), is(greaterThanOrEqualTo(1)));
140         assertThat(authzRange.min(), is(greaterThanOrEqualTo(0)));
141         assertThat(algsRange.min(), is(greaterThanOrEqualTo(1)));
142         assertThat(audiencesRange.min(), is(greaterThanOrEqualTo(1)));
143         assertThat(usersRange.min(), is(greaterThanOrEqualTo(1)));
144         assertThat(rolesRange.min(), is(greaterThanOrEqualTo(0)));
145         assertThat(jwtCacheSizeRange.min(), is(greaterThanOrEqualTo(0)));
146 
147         // Create JWT authc realms and mocked authz realms. Initialize each JWT realm, and test ensureInitialized() before and after.
148         final JwtRealmsService jwtRealmsService = this.generateJwtRealmsService(jwtRealmsServiceSettingsBuilder);
149         final int realmsCount = randomIntBetween(realmsRange.min(), realmsRange.max());
150         final List<Realm> allRealms = new ArrayList<>(); // authc and authz realms
151         this.jwtIssuerAndRealms = new ArrayList<>(realmsCount);
152         for (int i = 0; i < realmsCount; i++) {
153             final int authzCount = randomIntBetween(authzRange.min(), authzRange.max());
154             final int algsCount = randomIntBetween(algsRange.min(), algsRange.max());
155             final int audiencesCount = randomIntBetween(audiencesRange.min(), audiencesRange.max());
156             final int usersCount = randomIntBetween(usersRange.min(), usersRange.max());
157             final int rolesCount = randomIntBetween(rolesRange.min(), rolesRange.max());
158             final int jwtCacheSize = randomIntBetween(jwtCacheSizeRange.min(), jwtCacheSizeRange.max());
159 
160             final JwtIssuer jwtIssuer = this.createJwtIssuer(
161                 i,
162                 randomFrom(jwtRealmsService.getPrincipalClaimNames()),
163                 algsCount,
164                 audiencesCount,
165                 usersCount,
166                 rolesCount,
167                 createHttpsServer
168             );
169             // If HTTPS server was created in JWT issuer, any exception after that point requires closing it to avoid a thread pool leak
170             try {
171                 final JwtRealmSettingsBuilder realmSettingsBuilder = this.createJwtRealmSettingsBuilder(
172                     jwtIssuer,
173                     authzCount,
174                     jwtCacheSize
175                 );
176                 final JwtRealm jwtRealm = this.createJwtRealm(allRealms, jwtRealmsService, jwtIssuer, realmSettingsBuilder);
177 
178                 // verify exception before initialize()
179                 final Exception exception = expectThrows(IllegalStateException.class, jwtRealm::ensureInitialized);
180                 assertThat(exception.getMessage(), equalTo("Realm has not been initialized"));
181 
182                 final JwtIssuerAndRealm jwtIssuerAndRealm = new JwtIssuerAndRealm(jwtIssuer, jwtRealm, realmSettingsBuilder);
183                 this.jwtIssuerAndRealms.add(jwtIssuerAndRealm);
184             } catch (Throwable t) {
185                 jwtIssuer.close();
186                 throw t;
187             }
188         }
189         allRealms.forEach(realm -> realm.initialize(allRealms, this.licenseState)); // JWT realms and authz realms
190         this.jwtIssuerAndRealms.forEach(p -> p.realm.ensureInitialized()); // verify no exception after initialize()
191         return this.jwtIssuerAndRealms;
192     }

195         final int i,
196         final String principalClaimName,
197         final int algsCount,
198         final int audiencesCount,
199         final int userCount,
200         final int roleCount,
201         final boolean createHttpsServer
202     ) throws Exception {
203         final String issuer = "iss" + (i + 1) + "_" + randomIntBetween(0, 9999);
204         final List<String> audiences = IntStream.range(0, audiencesCount).mapToObj(j -> issuer + "_aud" + (j + 1)).toList();
205         final Map<String, User> users = JwtTestCase.generateTestUsersWithRoles(userCount, roleCount);
206         // Allow algorithm repeats, to cover testing of multiple JWKs for same algorithm
207         final JwtIssuer jwtIssuer = new JwtIssuer(issuer, audiences, principalClaimName, users, createHttpsServer);
208         final List<String> algorithms = randomOfMinMaxNonUnique(algsCount, algsCount, JwtRealmSettings.SUPPORTED_SIGNATURE_ALGORITHMS);
209         final boolean areHmacJwksOidcSafe = randomBoolean();
210         final List<JwtIssuer.AlgJwkPair> algAndJwks = JwtRealmTestCase.randomJwks(algorithms, areHmacJwksOidcSafe);
211         jwtIssuer.setJwks(algAndJwks, areHmacJwksOidcSafe);
212         return jwtIssuer;
213     }

240     protected JwtRealmSettingsBuilder createJwtRealmSettingsBuilder(final JwtIssuer jwtIssuer, final int authzCount, final int jwtCacheSize)
241         throws Exception {
242         final String authcRealmName = "realm_" + jwtIssuer.issuerClaimValue;
243         final String[] authzRealmNames = IntStream.range(0, authzCount).mapToObj(z -> authcRealmName + "_authz" + z).toArray(String[]::new);
244 
245         final ClientAuthenticationType clientAuthenticationType = randomFrom(ClientAuthenticationType.values());
246         final Settings.Builder authcSettings = Settings.builder()
247             .put(this.globalSettings)
248             .put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.ALLOWED_ISSUER), jwtIssuer.issuerClaimValue)
249             .put(
250                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.ALLOWED_SIGNATURE_ALGORITHMS),
251                 String.join(",", jwtIssuer.algorithmsAll)
252             )
253             .put(
254                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.ALLOWED_AUDIENCES),
255                 randomFrom(jwtIssuer.audiencesClaimValue)
256             );
257         authcSettings.put(
258             RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_PRINCIPAL.getClaim()),
259             jwtIssuer.principalClaimName
260         );
261         if ((ClientAuthenticationType.SHARED_SECRET != clientAuthenticationType) || (randomBoolean())) {
262             // always set "None", optionally set "SharedSecret" or let it get picked by default
263             authcSettings.put(
264                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLIENT_AUTHENTICATION_TYPE),
265                 clientAuthenticationType
266             );
267         }
268         if (randomBoolean()) {
269             // optionally allow default, or set -1 disabled or non-zero for enabled
270             authcSettings.put(
271                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.ALLOWED_CLOCK_SKEW),
272                 randomBoolean() ? "-1" : randomBoolean() ? "0" : randomIntBetween(1, 5) + randomFrom("s", "m", "h")
273             );
274         }
275         if (jwtIssuer.encodedJwkSetPkcPublic.isEmpty() == false) {
276             final String jwkSetPath; // file or HTTPS URL
277             if (jwtIssuer.httpsServer == null) {
278                 jwkSetPath = super.saveToTempFile("jwkset.", ".json", jwtIssuer.encodedJwkSetPkcPublic);
279             } else {
280                 authcSettings.putList(
281                     RealmSettings.getFullSettingKey(
282                         new RealmConfig.RealmIdentifier(JwtRealmSettings.TYPE, authcRealmName),
283                         SSLConfigurationSettings.CAPATH_SETTING_REALM
284                     ),
285                     JwtIssuerHttpsServer.CERT_PATH.toString()
286                 );
287                 jwkSetPath = jwtIssuer.httpsServer.url;
288             }
289             authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.PKC_JWKSET_PATH), jwkSetPath);
290         }
291         if (randomBoolean()) {
292             // principal claim name is required, but principal claim pattern is optional
293             authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_PRINCIPAL.getPattern()), "^(.*)$");
294         }
295         if (randomBoolean()) {
296             // groups claim name is optional
297             authcSettings.put(
298                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_GROUPS.getClaim()),
299                 authcRealmName + "_groups"
300             );
301             if (randomBoolean()) {
302                 // if groups claim name is set, groups claim pattern is optional
303                 authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_GROUPS.getPattern()), "^(.*)$");
304             }
305         }
306         if (randomBoolean()) {
307             // dn claim name is optional
308             authcSettings.put(
309                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_DN.getClaim()),
310                 authcRealmName + "_dn"
311             );
312             if (randomBoolean()) {
313                 // if dn claim name is set, dn claim pattern is optional
314                 authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_DN.getPattern()), "^(.*)$");
315             }
316         }
317         if (randomBoolean()) {
318             // mail claim name is optional
319             authcSettings.put(
320                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_MAIL.getClaim()),
321                 authcRealmName + "_mail"
322             );
323             if (randomBoolean()) {
324                 // if mail claim name is set, dn claim pattern is optional
325                 authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_MAIL.getPattern()), "^(.*)$");
326             }
327         }
328         if (randomBoolean()) {
329             // full name claim name is optional
330             authcSettings.put(
331                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_NAME.getClaim()),
332                 authcRealmName + "_name"
333             );
334             if (randomBoolean()) {
335                 // if full name claim name is set, name claim pattern is optional
336                 authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLAIMS_NAME.getPattern()), "^(.*)$");
337             }
338         }
339         if (randomBoolean()) {
340             // allow default to be picked, or explicitly set true or false
341             authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.POPULATE_USER_METADATA), randomBoolean());
342         }
343         if ((authzRealmNames.length != 0) || (randomBoolean())) {
344             // always set non-empty list, otherwise leave it out or optionally set value to an empty list
345             authcSettings.put(
346                 RealmSettings.getFullSettingKey(authcRealmName, DelegatedAuthorizationSettings.AUTHZ_REALMS.apply(JwtRealmSettings.TYPE)),
347                 String.join(",", authzRealmNames)
348             );
349         }
350 
351         // JWT cache (on/off controlled by jwtCacheSize)
352         if (randomBoolean()) {
353             authcSettings.put(
354                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.JWT_CACHE_TTL),
355                 randomIntBetween(10, 120) + randomFrom("s", "m", "h")
356             );
357         }
358         authcSettings.put(RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.JWT_CACHE_SIZE), jwtCacheSize);
359 
360         // JWT authc realm secure settings
361         final MockSecureSettings secureSettings = new MockSecureSettings();
362         if (jwtIssuer.algAndJwksHmac.isEmpty() == false) {
363             secureSettings.setString(
364                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.HMAC_JWKSET),
365                 jwtIssuer.encodedJwkSetHmac
366             );
367         }
368         if (jwtIssuer.encodedKeyHmacOidc != null) {
369             secureSettings.setString(
370                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.HMAC_KEY),
371                 jwtIssuer.encodedKeyHmacOidc
372             );
373         }
374         if (clientAuthenticationType.equals(ClientAuthenticationType.SHARED_SECRET)) {
375             // always set if type is "SharedSecret"
376             final String clientAuthenticationSharedSecret = randomAlphaOfLength(64);
377             secureSettings.setString(
378                 RealmSettings.getFullSettingKey(authcRealmName, JwtRealmSettings.CLIENT_AUTHENTICATION_SHARED_SECRET),
379                 clientAuthenticationSharedSecret
380             );
381         }
382         authcSettings.setSecureSettings(secureSettings);
383         return new JwtRealmSettingsBuilder(authcRealmName, authcSettings);
384     }

386     protected JwtRealm createJwtRealm(
387         final List<Realm> allRealms, // JWT realms and authz realms
388         final JwtRealmsService jwtRealmsService,
389         final JwtIssuer jwtIssuer,
390         final JwtRealmSettingsBuilder realmSettingsBuilder
391     ) {
392         final String authcRealmName = realmSettingsBuilder.name;
393         final Settings settings = realmSettingsBuilder.settingsBuilder.build();
394         final RealmConfig authcConfig = super.buildRealmConfig(JwtRealmSettings.TYPE, authcRealmName, settings, (allRealms.size() + 1));
395         final SSLService sslService = new SSLService(TestEnvironment.newEnvironment(settings));
396         final List<String> authzRealmNames = settings.getAsList(
397             RealmSettings.getFullSettingKey(authcRealmName, DelegatedAuthorizationSettings.AUTHZ_REALMS.apply(JwtRealmSettings.TYPE))
398         );
399         final UserRoleMapper userRoleMapper = super.buildRoleMapper(authzRealmNames.isEmpty() ? jwtIssuer.principals : Map.of());
400 
401         // If authz names is not set, register the users here in the JWT authc realm.
402         final JwtRealm jwtRealm = new JwtRealm(authcConfig, jwtRealmsService, sslService, userRoleMapper);
403         allRealms.add(jwtRealm);
404 
405         // If authz names is set, register the users here in one of the authz realms.
406         if (authzRealmNames.isEmpty() == false) {
407             final String selected = randomFrom(authzRealmNames);
408             for (final String authzRealmName : authzRealmNames) {
409                 final RealmConfig authzConfig = this.buildRealmConfig("authz", authzRealmName, Settings.EMPTY, allRealms.size() + 1);
410                 final MockLookupRealm authzRealm = new MockLookupRealm(authzConfig);
411                 if (authzRealmName.equals(selected)) {
412                     jwtIssuer.principals.values().forEach(authzRealm::registerUser);
413                 }
414                 allRealms.add(authzRealm);
415             }
416         }
417         return jwtRealm;
418     }

433     protected void doMultipleAuthcAuthzAndVerifySuccess(
434         final JwtRealm jwtRealm,
435         final User user,
436         final SecureString jwt,
437         final SecureString sharedSecret,
438         final MinMax jwtAuthcRange
439     ) throws Exception {
440         assertThat(jwtAuthcRange.min(), is(greaterThanOrEqualTo(1)));
441 
442         // Select one JWT authc Issuer/Realm pair. Select one test user, to use inside the authc test loop.
443         final List<JwtRealm> jwtRealmsList = this.jwtIssuerAndRealms.stream().map(p -> p.realm).toList();
444 
445         // Select different test JWKs from the JWT realm, and generate test JWTs for the test user. Run the JWT through the chain.
446         final int jwtAuthcRepeats = randomIntBetween(jwtAuthcRange.min(), jwtAuthcRange.max());
447         for (int authcRun = 1; authcRun <= jwtAuthcRepeats; authcRun++) {
448             // Create request with headers set
449             final ThreadContext requestThreadContext = super.createThreadContext(jwt, sharedSecret);
450             LOGGER.info("REQ[" + authcRun + "/" + jwtAuthcRepeats + "] HEADERS=" + requestThreadContext.getHeaders());
451 
452             // Loop through all authc/authz realms. Confirm a JWT authc realm recognizes and extracts the request headers.
453             JwtAuthenticationToken jwtAuthenticationToken = null;
454             for (final JwtRealm candidateJwtRealm : jwtRealmsList) {
455                 final AuthenticationToken authenticationToken = candidateJwtRealm.token(requestThreadContext);
456                 if (authenticationToken != null) {
457                     assertThat(authenticationToken, isA(JwtAuthenticationToken.class));
458                     jwtAuthenticationToken = (JwtAuthenticationToken) authenticationToken;
459                     break;
460                 }
461             }
462             assertThat(jwtAuthenticationToken, is(notNullValue()));
463             final String tokenPrincipal = jwtAuthenticationToken.principal();
464             final SecureString tokenJwt = jwtAuthenticationToken.getEndUserSignedJwt();
465             final SecureString tokenSecret = jwtAuthenticationToken.getClientAuthenticationSharedSecret();
466             assertThat(tokenPrincipal, is(notNullValue()));
467             if (tokenJwt.equals(jwt) == false) {
468                 assertThat(tokenJwt, is(equalTo(jwt)));
469             }
470             assertThat(tokenJwt, is(equalTo(jwt)));
471             if (tokenSecret != null) {
472                 if (tokenSecret.equals(sharedSecret) == false) {
473                     assertThat(tokenSecret, is(equalTo(sharedSecret)));
474                 }
475                 assertThat(tokenSecret, is(equalTo(sharedSecret)));
476             }
477             LOGGER.info("GOT TOKEN: principal=[" + tokenPrincipal + "], jwt=[" + tokenJwt + "], secret=[" + tokenSecret + "].");
478 
479             // Loop through all authc/authz realms. Confirm user is returned with expected principal and roles.
480             User authenticatedUser = null;
481             final List<String> realmAuthenticationResults = new ArrayList<>();
482             final List<String> realmUsageStats = new ArrayList<>();
483             final List<Exception> realmFailureExceptions = new ArrayList<>(jwtRealmsList.size());
484             try {
485                 for (final JwtRealm candidateJwtRealm : jwtRealmsList) {
486                     LOGGER.info("TRY AUTHC: expected=[" + jwtRealm.name() + "], candidate[" + candidateJwtRealm.name() + "].");
487                     final PlainActionFuture<AuthenticationResult<User>> authenticateFuture = PlainActionFuture.newFuture();
488                     try {
489                         candidateJwtRealm.authenticate(jwtAuthenticationToken, authenticateFuture);
490                         final AuthenticationResult<User> authenticationResult = authenticateFuture.actionGet();
491                         LOGGER.info("Authentication result with realm [{}]: [{}]", candidateJwtRealm.name(), authenticationResult);
492                         final Exception authenticationResultException = authenticationResult.getException();
493                         final String realmResult = "  realms=["
494                             + jwtRealmsList.size()
495                             + "], expected=["
496                             + jwtRealm.name()
497                             + ","
498                             + jwtRealm.order()
499                             + "], current["
500                             + candidateJwtRealm.name()
501                             + ","
502                             + candidateJwtRealm.order()
503                             + "], status=["
504                             + authenticationResult.getStatus()
505                             + "], authenticated=["
506                             + authenticationResult.isAuthenticated()
507                             + "], msg=["
508                             + authenticationResult.getMessage()
509                             + "], meta=["
510                             + authenticationResult.getMetadata()
511                             + "], user=["
512                             + authenticationResult.getValue()
513                             + "].";
514                         realmAuthenticationResults.add(realmResult);
515                         realmFailureExceptions.add(new Exception(realmResult, authenticationResultException));
516                         switch (authenticationResult.getStatus()) {
517                             case SUCCESS:
518                                 assertThat("Unexpected realm SUCCESS status", candidateJwtRealm.name(), is(equalTo(jwtRealm.name())));
519                                 assertThat("Expected realm authc false", authenticationResult.isAuthenticated(), is(equalTo(true)));
520                                 assertThat("Expected realm exception thrown", authenticationResult.getException(), is(nullValue()));
521                                 assertThat("Expected realm message null", authenticationResult.getMessage(), is(nullValue()));
522                                 assertThat("Expected realm metadata empty", authenticationResult.getMetadata(), is(anEmptyMap()));
523                                 authenticatedUser = authenticationResult.getValue();
524                                 assertThat("Expected realm user null", authenticatedUser, is(notNullValue()));
525                                 break;
526                             case CONTINUE:
527                                 assertThat("Expected realm CONTINUE status", candidateJwtRealm.name(), is(not(equalTo(jwtRealm.name()))));
528                                 assertThat("Unexpected realm authc success", authenticationResult.isAuthenticated(), is(equalTo(false)));
529                                 continue;
530                             case TERMINATE:
531                                 assertThat("Expected realm TERMINATE status", candidateJwtRealm.name(), is(not(equalTo(jwtRealm.name()))));
532                                 assertThat("Unexpected realm authc success", authenticationResult.isAuthenticated(), is(equalTo(false)));
533                                 break;
534                             default:
535                                 fail("Unexpected AuthenticationResult.Status=[" + authenticationResult.getStatus() + "]");
536                                 break;
537                         }
538                         break; // Only SUCCESS falls through to here, break out of the loop
539                     } catch (Exception e) {
540                         realmFailureExceptions.add(e);
541                         throw e;
542                     } finally {
543                         final PlainActionFuture<Map<String, Object>> usageStatsFuture = PlainActionFuture.newFuture();
544                         candidateJwtRealm.usageStats(usageStatsFuture);
545                         realmUsageStats.add(
546                             "   realm["
547                                 + candidateJwtRealm.name()
548                                 + ","
549                                 + candidateJwtRealm.order()
550                                 + "/"
551                                 + jwtRealmsList.size()
552                                 + "], stats=["
553                                 + usageStatsFuture.actionGet()
554                                 + "]"
555                         );
556                     }
557                 }
558                 // Loop ended. Confirm user is returned with expected principal and roles.
559                 assertThat("Expected realm " + jwtRealm.name() + " to authenticate.", authenticatedUser, is(notNullValue()));
560                 assertThat(user.principal(), equalTo(authenticatedUser.principal()));
561                 assertThat(new TreeSet<>(Arrays.asList(user.roles())), equalTo(new TreeSet<>(Arrays.asList(authenticatedUser.roles()))));
562                 if (jwtRealm.delegatedAuthorizationSupport.hasDelegation()) {
563                     assertThat(user.metadata(), is(equalTo(authenticatedUser.metadata()))); // delegated authz returns user's metadata
564                 } else if (jwtRealm.populateUserMetadata) {
565                     assertThat(authenticatedUser.metadata(), is(not(anEmptyMap()))); // role mapping with flag true returns non-empty
566                 } else {
567                     assertThat(authenticatedUser.metadata(), is(anEmptyMap())); // role mapping with flag false returns empty
568                 }
569             } catch (Throwable t) {
570                 realmFailureExceptions.forEach(t::addSuppressed); // all previous realm exceptions
571                 // LOGGER.error("Unexpected exception.", t);
572                 throw t;
573             } finally {
574                 LOGGER.info("STATS: expected=[" + jwtRealm.name() + "]\n" + String.join("\n", realmUsageStats));
575                 if (authenticatedUser != null) {
576                     LOGGER.info("RESULT: expected=[" + jwtRealm.name() + "]\n" + String.join("\n", realmAuthenticationResults));
577                 }
578             }
579         }
580         LOGGER.info("Test succeeded");
581     }

589     protected SecureString randomJwt(final JwtIssuerAndRealm jwtIssuerAndRealm, User user) throws Exception {
590         final JwtIssuer.AlgJwkPair algJwkPair = randomFrom(jwtIssuerAndRealm.issuer.algAndJwksAll);
591         final JWK jwk = algJwkPair.jwk();
592         LOGGER.info(
593             "ALG["
594                 + algJwkPair.alg()
595                 + "]. JWK: kty=["
596                 + jwk.getKeyType()
597                 + "], len=["
598                 + jwk.size()
599                 + "], alg=["
600                 + jwk.getAlgorithm()
601                 + "], use=["
602                 + jwk.getKeyUse()
603                 + "], ops=["
604                 + jwk.getKeyOperations()
605                 + "], kid=["
606                 + jwk.getKeyID()
607                 + "]."
608         );
609 
610         final Instant now = Instant.now().truncatedTo(ChronoUnit.SECONDS);
611         final SignedJWT unsignedJwt = JwtTestCase.buildUnsignedJwt(
612             randomBoolean() ? null : JOSEObjectType.JWT.toString(), // kty
613             randomBoolean() ? null : jwk.getKeyID(), // kid
614             algJwkPair.alg(), // alg
615             randomAlphaOfLengthBetween(10, 20), // jwtID
616             jwtIssuerAndRealm.realm.allowedIssuer, // iss
617             jwtIssuerAndRealm.realm.allowedAudiences, // aud
618             randomBoolean() ? null : randomBoolean() ? user.principal() : user.principal() + "_" + randomInt(9), // sub claim value
619             jwtIssuerAndRealm.realm.claimParserPrincipal.getClaimName(), // principal claim name
620             user.principal(), // principal claim value
621             jwtIssuerAndRealm.realm.claimParserGroups.getClaimName(), // group claim name
622             List.of(user.roles()), // group claim value
623             Date.from(now.minusSeconds(60 * randomLongBetween(10, 20))), // auth_time
624             Date.from(now.minusSeconds(randomBoolean() ? 0 : 60 * randomLongBetween(5, 10))), // iat
625             Date.from(now), // nbf
626             Date.from(now.plusSeconds(60 * randomLongBetween(3600, 7200))), // exp
627             randomBoolean() ? null : new Nonce(32).toString(),
628             randomBoolean() ? null : Map.of("other1", randomAlphaOfLength(10), "other2", randomAlphaOfLength(10))
629         );
630         final SecureString signedJWT = JwtValidateUtil.signJwt(jwk, unsignedJwt);
631         assertThat(JwtValidateUtil.verifyJwt(jwk, SignedJWT.parse(signedJWT.toString())), is(equalTo(true)));
632         return signedJWT;
633     }
