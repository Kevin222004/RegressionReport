40      private static final Map<Class<? extends Entry<?>>, Function<XContentParser, ? extends Entry<?>>> ENTRY_PARSERS = new HashMap<>();
41      static {
42          ENTRY_PARSERS.put(TermSuggestion.Entry.class, TermSuggestion.Entry::fromXContent);
43          ENTRY_PARSERS.put(PhraseSuggestion.Entry.class, PhraseSuggestion.Entry::fromXContent);
44          ENTRY_PARSERS.put(CompletionSuggestion.Entry.class, CompletionSuggestion.Entry::fromXContent);
45      }

47      /**
48       * Create a randomized Suggestion.Entry
49       */
50      @SuppressWarnings("unchecked")
51      public static <O extends Option> Entry<O> createTestItem(Class<? extends Entry<O>> entryType) {
52          Text entryText = new Text(randomAlphaOfLengthBetween(5, 15));
53          int offset = randomInt();
54          int length = randomInt();
55          Entry<O> entry;
56          Supplier<Option> supplier;
57          if (entryType == (Class<? extends Entry<? extends Option>>) TermSuggestion.Entry.class) {
58              entry = (Entry<O>) new TermSuggestion.Entry(entryText, offset, length);
59              supplier = TermSuggestionOptionTests::createTestItem;
60          } else if (entryType == (Class<? extends Entry<? extends Option>>) PhraseSuggestion.Entry.class) {
61              entry = (Entry<O>) new PhraseSuggestion.Entry(entryText, offset, length, randomDouble());
62              supplier = SuggestionOptionTests::createTestItem;
63          } else if (entryType == (Class<? extends Entry<? extends Option>>) CompletionSuggestion.Entry.class) {
64              entry = (Entry<O>) new CompletionSuggestion.Entry(entryText, offset, length);
65              supplier = CompletionSuggestionOptionTests::createTestItem;
66          } else {
67              throw new UnsupportedOperationException("entryType not supported [" + entryType + "]");
68          }
69          int numOptions = randomIntBetween(0, 5);
70          for (int i = 0; i < numOptions; i++) {
71              entry.addOption((O) supplier.get());
72          }
73          return entry;
74      }

76      public void testFromXContent() throws IOException {
77          doTestFromXContent(false);
78      }
79  
80      public void testFromXContentWithRandomFields() throws IOException {
81          doTestFromXContent(true);
82      }
83  
84      @SuppressWarnings("unchecked")
85      private void doTestFromXContent(boolean addRandomFields) throws IOException {
86          for (Class<? extends Entry<?>> entryType : ENTRY_PARSERS.keySet()) {
87              Entry<Option> entry = createTestItem((forciblyCast(entryType)));
88              XContentType xContentType = randomFrom(XContentType.values());
89              boolean humanReadable = randomBoolean();
90              BytesReference originalBytes = toShuffledXContent(entry, xContentType, ToXContent.EMPTY_PARAMS, humanReadable);
91              BytesReference mutated;
92              if (addRandomFields) {
93                  // "contexts" is an object consisting of key/array pairs, we shouldn't add anything random there
94                  // also there can be inner search hits fields inside this option, we need to exclude another couple of paths
95                  // where we cannot add random stuff
96                  // exclude "options" which contain SearchHits,
97                  // on root level of SearchHit fields are interpreted as meta-fields and will be kept
98                  Predicate<String> excludeFilter = (path -> path.endsWith(CompletionSuggestion.Entry.Option.CONTEXTS.getPreferredName())
99                      || path.endsWith("highlight")
100                     || path.contains("fields")
101                     || path.contains("_source")
102                     || path.contains("inner_hits")
103                     || path.contains("options"));
104 
105                 mutated = insertRandomFields(xContentType, originalBytes, excludeFilter, random());
106             } else {
107                 mutated = originalBytes;
108             }
109             Entry<Option> parsed;
110             try (XContentParser parser = createParser(xContentType.xContent(), mutated)) {
111                 ensureExpectedToken(XContentParser.Token.START_OBJECT, parser.nextToken(), parser);
112                 parsed = (Entry<Option>) ENTRY_PARSERS.get(entry.getClass()).apply(parser);
113                 assertEquals(XContentParser.Token.END_OBJECT, parser.currentToken());
114                 assertNull(parser.nextToken());
115             }
116             assertEquals(entry.getClass(), parsed.getClass());
117             assertEquals(entry.getText(), parsed.getText());
118             assertEquals(entry.getLength(), parsed.getLength());
119             assertEquals(entry.getOffset(), parsed.getOffset());
120             assertEquals(entry.getOptions().size(), parsed.getOptions().size());
121             for (int i = 0; i < entry.getOptions().size(); i++) {
122                 assertEquals(entry.getOptions().get(i).getClass(), parsed.getOptions().get(i).getClass());
123             }
124             assertToXContentEquivalent(originalBytes, toXContent(parsed, xContentType, humanReadable), xContentType);
125         }
126     }

128     public void testToXContent() throws IOException {
129         PhraseSuggestion.Entry.Option phraseOption = new PhraseSuggestion.Entry.Option(
130             new Text("someText"),
131             new Text("somethingHighlighted"),
132             1.3f,
133             true
134         );
135         PhraseSuggestion.Entry phraseEntry = new PhraseSuggestion.Entry(new Text("entryText"), 42, 313);
136         phraseEntry.addOption(phraseOption);
137         BytesReference xContent = toXContent(phraseEntry, XContentType.JSON, randomBoolean());
138         assertEquals(XContentHelper.stripWhitespace("""
139             {
140               "text": "entryText",
141               "offset": 42,
142               "length": 313,
143               "options": [
144                 {
145                   "text": "someText",
146                   "highlighted": "somethingHighlighted",
147                   "score": 1.3,
148                   "collate_match": true
149                 }
150               ]
151             }"""), xContent.utf8ToString());
152 
153         TermSuggestion.Entry.Option termOption = new TermSuggestion.Entry.Option(new Text("termSuggestOption"), 42, 3.13f);
154         TermSuggestion.Entry termEntry = new TermSuggestion.Entry(new Text("entryText"), 42, 313);
155         termEntry.addOption(termOption);
156         xContent = toXContent(termEntry, XContentType.JSON, randomBoolean());
157         assertEquals(XContentHelper.stripWhitespace("""
158             {
159               "text": "entryText",
160               "offset": 42,
161               "length": 313,
162               "options": [ { "text": "termSuggestOption", "score": 3.13, "freq": 42 } ]
163             }"""), xContent.utf8ToString());
164 
165         CompletionSuggestion.Entry.Option completionOption = new CompletionSuggestion.Entry.Option(
166             -1,
167             new Text("completionOption"),
168             3.13f,
169             Collections.singletonMap("key", Collections.singleton("value"))
170         );
171         CompletionSuggestion.Entry completionEntry = new CompletionSuggestion.Entry(new Text("entryText"), 42, 313);
172         completionEntry.addOption(completionOption);
173         xContent = toXContent(completionEntry, XContentType.JSON, randomBoolean());
174         assertEquals(XContentHelper.stripWhitespace("""
175             {
176               "text": "entryText",
177               "offset": 42,
178               "length": 313,
179               "options": [
180                 {
181                   "text": "completionOption",
182                   "score": 3.13,
183                   "contexts": {
184                     "key": [ "value" ]
185                   }
186                 }
187               ]
188             }"""), xContent.utf8ToString());
189     }
