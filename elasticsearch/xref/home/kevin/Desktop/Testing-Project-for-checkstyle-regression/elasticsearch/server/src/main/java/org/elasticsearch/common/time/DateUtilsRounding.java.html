19  /**
20   * This class has been copied from different locations within the joda time package, as
21   * these methods fast when used for rounding, as they do not require conversion to java
22   * time objects
23   *
24   * This code has been copied from jodatime 2.10.1
25   * The source can be found at https://github.com/JodaOrg/joda-time/tree/v2.10.1
26   *
27   * See following methods have been copied (along with required helper variables)
28   *
29   * - org.joda.time.chrono.GregorianChronology.calculateFirstDayOfYearMillis(int year)
30   * - org.joda.time.chrono.BasicChronology.getYear(int year)
31   * - org.joda.time.chrono.BasicGJChronology.getMonthOfYear(long utcMillis, int year)
32   * - org.joda.time.chrono.BasicGJChronology.getTotalMillisByYearMonth(int year, int month)
33   */
34  class DateUtilsRounding {
35  
36      private static final int DAYS_0000_TO_1970 = 719527;
37      private static final int MILLIS_PER_DAY = 86_400_000;
38      private static final long MILLIS_PER_YEAR = 31556952000L;
39  
40      // see org.joda.time.chrono.BasicGJChronology
41      private static final long[] MIN_TOTAL_MILLIS_BY_MONTH_ARRAY;
42      private static final long[] MAX_TOTAL_MILLIS_BY_MONTH_ARRAY;
43      private static final int[] MIN_DAYS_PER_MONTH_ARRAY = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
44      private static final int[] MAX_DAYS_PER_MONTH_ARRAY = { 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
45  
46      static {
47          MIN_TOTAL_MILLIS_BY_MONTH_ARRAY = new long[12];
48          MAX_TOTAL_MILLIS_BY_MONTH_ARRAY = new long[12];
49  
50          long minSum = 0;
51          long maxSum = 0;
52          for (int i = 0; i < 11; i++) {
53              long millis = MIN_DAYS_PER_MONTH_ARRAY[i] * (long) MILLIS_PER_DAY;
54              minSum += millis;
55              MIN_TOTAL_MILLIS_BY_MONTH_ARRAY[i + 1] = minSum;
56  
57              millis = MAX_DAYS_PER_MONTH_ARRAY[i] * (long) MILLIS_PER_DAY;
58              maxSum += millis;
59              MAX_TOTAL_MILLIS_BY_MONTH_ARRAY[i + 1] = maxSum;
60          }
61      }

63      /**
64       * calculates the first day of a year in milliseconds since the epoch (assuming UTC)
65       *
66       * @param year the year
67       * @return the milliseconds since the epoch of the first of january at midnight of the specified year
68       */
69      // see org.joda.time.chrono.GregorianChronology.calculateFirstDayOfYearMillis
70      static long utcMillisAtStartOfYear(final int year) {
71          // Initial value is just temporary.
72          int leapYears = year / 100;
73          if (year < 0) {
74              // Add 3 before shifting right since /4 and >>2 behave differently
75              // on negative numbers. When the expression is written as
76              // (year / 4) - (year / 100) + (year / 400),
77              // it works for both positive and negative values, except this optimization
78              // eliminates two divisions.
79              leapYears = ((year + 3) >> 2) - leapYears + ((leapYears + 3) >> 2) - 1;
80          } else {
81              leapYears = (year >> 2) - leapYears + (leapYears >> 2);
82              if (isLeapYear(year)) {
83                  leapYears--;
84              }
85          }
86  
87          return (year * 365L + (leapYears - DAYS_0000_TO_1970)) * MILLIS_PER_DAY; // millis per day
88      }
89  
90      static boolean isLeapYear(final int year) {
91          // Joda had
92          // return ((year & 3) == 0) && ((year % 100) != 0 || (year % 400) == 0);
93          // But we've replaced that with this:
94          if ((year & 3) != 0) {
95              return false;
96          }
97          if (year % 100 != 0) {
98              return true;
99          }
100         return ((year / 100) & 3) == 0;
101         /*
102          * It is a little faster because it saves a division. We don't have good
103          * measurements for this method on its own, but this change speeds up
104          * rounding the nearest month by about 8%.
105          *
106          * Note: If you decompile this method to x86 assembly you won't see the
107          * division you'd expect from % 100 and / 100. Instead you'll see a funny
108          * sequence of bit twiddling operations which the jvm thinks is faster.
109          * Division is slow so it almost certainly is.
110          */
111     }
112 
113     private static final long AVERAGE_MILLIS_PER_YEAR_DIVIDED_BY_TWO = MILLIS_PER_YEAR / 2;
114     private static final long APPROX_MILLIS_AT_EPOCH_DIVIDED_BY_TWO = (1970L * MILLIS_PER_YEAR) / 2;
115 
116     // see org.joda.time.chrono.BasicChronology
117     static int getYear(final long utcMillis) {
118         // Get an initial estimate of the year, and the millis value that
119         // represents the start of that year. Then verify estimate and fix if
120         // necessary.
121 
122         // Initial estimate uses values divided by two to avoid overflow.
123         long unitMillis = AVERAGE_MILLIS_PER_YEAR_DIVIDED_BY_TWO;
124         long i2 = (utcMillis >> 1) + APPROX_MILLIS_AT_EPOCH_DIVIDED_BY_TWO;
125         if (i2 < 0) {
126             i2 = i2 - unitMillis + 1;
127         }
128         int year = (int) (i2 / unitMillis);
129 
130         long yearStart = utcMillisAtStartOfYear(year);
131         long diff = utcMillis - yearStart;
132 
133         if (diff < 0) {
134             year--;
135         } else if (diff >= MILLIS_PER_DAY * 365L) {
136             // One year may need to be added to fix estimate.
137             long oneYear;
138             if (isLeapYear(year)) {
139                 oneYear = MILLIS_PER_DAY * 366L;
140             } else {
141                 oneYear = MILLIS_PER_DAY * 365L;
142             }
143 
144             yearStart += oneYear;
145 
146             if (yearStart <= utcMillis) {
147                 // Didn't go too far, so actually add one year.
148                 year++;
149             }
150         }
151 
152         return year;
153     }
154 
155     // see org.joda.time.chrono.BasicGJChronology
156     static int getMonthOfYear(final long utcMillis, final int year) {
157         // Perform a binary search to get the month. To make it go even faster,
158         // compare using ints instead of longs. The number of milliseconds per
159         // year exceeds the limit of a 32-bit int's capacity, so divide by
160         // 1024. No precision is lost (except time of day) since the number of
161         // milliseconds per day contains 1024 as a factor. After the division,
162         // the instant isn't measured in milliseconds, but in units of
163         // (128/125)seconds.
164 
165         int i = (int) ((utcMillis - utcMillisAtStartOfYear(year)) >> 10);
166 
167         // There are 86400000 milliseconds per day, but divided by 1024 is
168         // 84375. There are 84375 (128/125)seconds per day.
169 
170         return (isLeapYear(year))
171             ? ((i < 182 * 84375)
172                 ? ((i < 91 * 84375)
173                     ? ((i < 31 * 84375) ? 1 : (i < 60 * 84375) ? 2 : 3)
174                     : ((i < 121 * 84375) ? 4 : (i < 152 * 84375) ? 5 : 6))
175                 : ((i < 274 * 84375)
176                     ? ((i < 213 * 84375) ? 7 : (i < 244 * 84375) ? 8 : 9)
177                     : ((i < 305 * 84375) ? 10 : (i < 335 * 84375) ? 11 : 12)))
178             : ((i < 181 * 84375)
179                 ? ((i < 90 * 84375)
180                     ? ((i < 31 * 84375) ? 1 : (i < 59 * 84375) ? 2 : 3)
181                     : ((i < 120 * 84375) ? 4 : (i < 151 * 84375) ? 5 : 6))
182                 : ((i < 273 * 84375)
183                     ? ((i < 212 * 84375) ? 7 : (i < 243 * 84375) ? 8 : 9)
184                     : ((i < 304 * 84375) ? 10 : (i < 334 * 84375) ? 11 : 12)));
185     }
186 
187     // see org.joda.time.chrono.BasicGJChronology
188     static long getTotalMillisByYearMonth(final int year, final int month) {
189         if (isLeapYear(year)) {
190             return MAX_TOTAL_MILLIS_BY_MONTH_ARRAY[month - 1];
191         } else {
192             return MIN_TOTAL_MILLIS_BY_MONTH_ARRAY[month - 1];
193         }
194     }
