120     public void setRefid(Reference r) throws BuildException {
121         if (dir != null || defaultPatterns.hasPatterns(getProject())) {
122             throw tooManyAttributes();

233     /**
234      * Creates a single file fileset.
235      * @param file the single <code>File</code> included in this
236      *             <code>AbstractFileSet</code>.
237      */
238     public synchronized void setFile(File file) {
239         if (isReference()) {
240             throw tooManyAttributes();
241         }
242         if (fileAttributeUsed) {
243             if (getDir().equals(file.getParentFile())) {
244                 String[] includes = defaultPatterns.getIncludePatterns(getProject());
245                 if (includes.length == 1 && includes[0].equals(file.getName())) {
246                     // NOOP, setFile has been invoked twice with the same parameter
247                     return;
248                 }
249             }
250             throw new BuildException("setFile cannot be called twice with different arguments");
251         } else if (getDir() != null) {
252             throw dirAndFileAreMutuallyExclusive();
253         }
254         setDir(file.getParentFile());
255         fileAttributeUsed = true;
256         createInclude().setName(file.getName());
257     }

492     /**
493      * Returns the directory scanner needed to access the files to process.
494      * @param p the Project against which the DirectoryScanner should be configured.
495      * @return a <code>DirectoryScanner</code> instance.
496      */
497     public DirectoryScanner getDirectoryScanner(Project p) {
498         if (isReference()) {
499             return getRef(p).getDirectoryScanner(p);
500         }
501         dieOnCircularReference();
502         final DirectoryScanner ds;
503         synchronized (this) {
504             if (directoryScanner != null && p == getProject()) {
505                 ds = directoryScanner;
506             } else {
507                 if (dir == null) {
508                     throw new BuildException("No directory specified for %s.",
509                         getDataTypeName());
510                 }
511                 if (!dir.exists() && errorOnMissingDir) {
512                     throw new BuildException(dir.getAbsolutePath()
513                                              + DirectoryScanner
514                                              .DOES_NOT_EXIST_POSTFIX);
515                 }
516                 if (!dir.isDirectory() && dir.exists()) {
517                     throw new BuildException("%s is not a directory.",
518                         dir.getAbsolutePath());
519                 }
520                 ds = new DirectoryScanner();
521                 setupDirectoryScanner(ds, p);
522                 ds.setFollowSymlinks(followSymlinks);
523                 ds.setErrorOnMissingDir(errorOnMissingDir);
524                 ds.setMaxLevelsOfSymlinks(maxLevelsOfSymlinks);
525                 directoryScanner = (p == getProject()) ? ds : directoryScanner;
526             }
527         }
528         ds.scan();
529         return ds;
530     }

541     /**
542      * Set up the specified directory scanner against the specified project.
543      * @param ds a <code>FileScanner</code> instance.
544      * @param p an Ant <code>Project</code> instance.
545      */
546     public synchronized void setupDirectoryScanner(FileScanner ds, Project p) {
547         if (isReference()) {
548             getRef(p).setupDirectoryScanner(ds, p);
549             return;
550         }
551         dieOnCircularReference(p);
552         if (ds == null) {
553             throw new IllegalArgumentException("ds cannot be null");
554         }
555         ds.setBasedir(dir);
556 
557         PatternSet ps = mergePatterns(p);
558         p.log(getDataTypeName() + ": Setup scanner in dir " + dir
559             + " with " + ps, Project.MSG_DEBUG);
560 
561         ds.setIncludes(ps.getIncludePatterns(p));
562         ds.setExcludes(ps.getExcludePatterns(p));
563         if (ds instanceof SelectorScanner) {
564             SelectorScanner ss = (SelectorScanner) ds;
565             ss.setSelectors(getSelectors(p));
566         }
567         if (useDefaultExcludes) {
568             ds.addDefaultExcludes();
569         }
570         ds.setCaseSensitive(caseSensitive);
571     }

903     @Override

912     /**
913      * Creates a deep clone of this instance, except for the nested
914      * selectors (the list of selectors is a shallow clone of this
915      * instance's list).
916      * @return the cloned object
917      * @since Ant 1.6
918      */
919     @Override
920     public synchronized Object clone() {
921         if (isReference()) {
922             return (getRef(getProject())).clone();
923         }
924         try {
925             AbstractFileSet fs = (AbstractFileSet) super.clone();
926             fs.defaultPatterns = (PatternSet) defaultPatterns.clone();
927             fs.additionalPatterns = additionalPatterns.stream().map(PatternSet::clone)
928                     .map(PatternSet.class::cast).collect(Collectors.toList());
929             fs.selectors = new ArrayList<>(selectors);
930             return fs;
931         } catch (CloneNotSupportedException e) {
932             throw new BuildException(e);
933         }
934     }

960     /**
961      * Get the merged patterns for this AbstractFileSet.
962      * @param p the project to use.
963      * @return the default patternset merged with the additional sets
964      * in a new PatternSet instance.
965      *
966      * @since Ant 1.7
967      */
968     public synchronized PatternSet mergePatterns(Project p) {
969         if (isReference()) {
970             return getRef(p).mergePatterns(p);
971         }
972         dieOnCircularReference();
973         PatternSet ps = (PatternSet) defaultPatterns.clone();
974         additionalPatterns.forEach(pat -> ps.append(pat, p));
975         return ps;
976     }

978     @Override
979     protected synchronized void dieOnCircularReference(Stack<Object> stk, Project p)
980         throws BuildException {
981         if (isChecked()) {
982             return;
983         }
984         if (isReference()) {
985             super.dieOnCircularReference(stk, p);
986         } else {
987             selectors.stream().filter(DataType.class::isInstance).map(DataType.class::cast)
988                     .forEach(type -> pushAndInvokeCircularReferenceCheck(type, stk, p));
989             additionalPatterns.forEach(ps -> pushAndInvokeCircularReferenceCheck(ps, stk, p));
990             setChecked(true);
991         }
992     }
